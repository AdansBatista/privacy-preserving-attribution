% Pandoc's default template calls \xmpquote{...} inside the PDF metadata
% (pdfkeywords). That macro comes from hyperxmp, but hyperxmp must load AFTER
% hyperref — and header-includes runs before hyperref. Simplest workaround:
% define \xmpquote as a pass-through so the keywords still get into the PDF
% metadata without XMP quoting (which is a nice-to-have, not load-bearing).
\providecommand{\xmpquote}[1]{#1}

% Tighter spacing around display math for readability in long proofs.
\AtBeginDocument{%
  \setlength{\abovedisplayskip}{6pt plus 2pt minus 2pt}%
  \setlength{\belowdisplayskip}{6pt plus 2pt minus 2pt}%
}

% Soft-wrap long URLs inside footnotes.
\usepackage{xurl}

% Nicer section titles for a long paper.
\usepackage[compact]{titlesec}
\titlespacing*{\section}{0pt}{1.2em}{0.5em}
\titlespacing*{\subsection}{0pt}{1.0em}{0.4em}
\titlespacing*{\subsubsection}{0pt}{0.8em}{0.3em}

% --- SSRN-style title page (matches standard economics-journal presentation) ---
\usepackage{titling}

% Title: small caps, larger, centered, with room around it.
\pretitle{\begin{center}\LARGE\scshape}
\posttitle{\par\end{center}\vskip 1.5em}

% Author: centered, normal size, footnote affiliation via \thanks{}.
\preauthor{\begin{center}\large}
\postauthor{\par\end{center}\vskip 0.5em}

% Date: centered, smaller.
\predate{\begin{center}\normalsize}
\postdate{\par\end{center}\vskip 1.5em}

% Abstract: indented both sides, bold centered "Abstract" label, slightly
% smaller body text — matches the standard SSRN/economics-paper idiom.
\usepackage{changepage}
\renewenvironment{abstract}{%
  \begin{center}%
    \normalsize\textbf{Abstract}%
  \end{center}%
  \begin{adjustwidth}{2em}{2em}%
  \small\noindent%
}{%
  \end{adjustwidth}%
  \vskip 1em%
}

% Put the TOC on its own page, starting fresh after the title page.
\usepackage{etoolbox}
\pretocmd{\tableofcontents}{\clearpage}{}{}
\apptocmd{\tableofcontents}{\clearpage}{}{}
